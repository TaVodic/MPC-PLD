Z následujících ukázek syntetizovatelného VHDL kódu vyberte ty, které obsahují alespoň jednu chybu. V ukázkách nehledejte syntaktické chyby (chybějící středníky apod.), ale chyby sémantické, latche a další nedovolené konstrukce pro syntézu.

Úloha 1 Vyberte jednu nebo více možností:
process(clk)
begin
  if rising_edge(clk) then
    cnt_ena <= not cnt_ena;     -- cnt_ena is signal
  end if;
end process;
process(clk)
begin
  if rising_edge(clk) then
    led_o <= led_green;
  end if;
end process;

process(clk)
begin
  if rising_edge(clk) then
    led_green <= not led_green;
  end if;
end process;
process(clk, cnt_ena)
begin
  if clk = '1' then
    led_o <= cnt_ena;
  end if;
end process;
entity uart_gen is
  port (
    clk  : in  std_logic;
    tx_o : out std_logic := '1' -- '1' is initial output value
  );
end uart_gen;

architecture uart_gen_arch of uart_gen is
  signal tx : std_logic;
begin
  ...
  tx_o <= tx and tx_o;
end architecture uart_gen_arch;
architecture top_arch of top is
  ...
begin
  led_o <= led_green_i when a_i(0) = '1' else

architecture top_arch of top is
begin
  led_o <= (others => '1');  -- default assignment

  led_o (0) <= '1' when sw_i = "11" else

Úloha 5 Vyberte jednu nebo více možností:
a.

result <= ("1000" & a) + b;

b.

c.

d.

result <= ("0000" & a) + b;
